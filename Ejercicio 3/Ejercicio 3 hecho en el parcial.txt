.data:
    notas: .byte 0xA 0xB0 0xF1 0x07
    largo: .byte 4
.text:
    la s0, notas
    lb s1, zero, largo
    jal ra, mitad
    jal ra, comenzar
    slt a0, a1, a0
comenzar: addi t0, zero, 0
    addi t1, zero, 0
while: bgt t0, s1, return
    addi t2, zero, 0(s0)
    andi t2, t2, 0xFF
    slti t3, t2, 0xA0
    add t1, t1, t3
    addi t0, t0, 1
    addi s0, s0, 1
    add a0, zero, t1
    j while
mitad: srli a1, s1, 1
    j return
return: ret
